DROP TABLE tblDataHarvester_Config

CREATE TABLE tblDataHarvester_Config (
    DHC_RecordID INT IDENTITY PRIMARY KEY,
    DHC_Name NVARCHAR(64),
    DHC_Symbol NVARCHAR(64),
    DHC_Exchange NVARCHAR(64),
    DHC_Interval NVARCHAR(16),
    DHC_isActive BIT,
    DHC_LastSuccess DATETIME,
    DHC_RetryMax INT DEFAULT 3,
    DHC_CreatedAt DATETIME DEFAULT GETDATE()
);

DROP TABLE tblDataHarvester

CREATE TABLE tblDataHarvester (
    DH_RecordID UNIQUEIDENTIFIER DEFAULT NEWSEQUENTIALID() PRIMARY KEY,
    DH_DHC_RecordID INT,
    DH_TimeStamp DATETIME,
    DH_Open FLOAT,
    DH_High FLOAT,
    DH_Low FLOAT,
    DH_Close FLOAT,
    DH_Volume FLOAT,
    DH_CloseChange FLOAT,        -- Close minus previous bar's Close
    DH_CloseChangePct FLOAT,     -- CloseChange as a percentage of previous bar's Close
    DH_CreatedAt DATETIME DEFAULT GETDATE(),
    CONSTRAINT uq_dhcrec_time UNIQUE(DH_DHC_RecordID, DH_TimeStamp)
);

DROP TABLE tblDataHarvester_Log

CREATE TABLE tblDataHarvester_Log (
    DHL_RecordID INT IDENTITY PRIMARY KEY,
    DHL_DHC_RecordID NVARCHAR(50),
    DHL_Status NVARCHAR(20),
    DHL_Message NVARCHAR(MAX),
    DHL_CreatedAt DATETIME DEFAULT GETDATE()
);

CREATE VIEW dbo.viwDataHarvester AS
SELECT DHC_RecordID [ID],DHC_Name [Name],DHC_Symbol [Symbol],DHC_Exchange [Exchange],DHC_Interval [Interval]
,DH_TimeStamp [DateTime],CAST(DH_TimeStamp AS DATE) [Date], CONVERT(DATE,LEFT(CAST(DH_TimeStamp AS DATE),7)+'-01') [YM1]
,DH_Open [Open],DH_High [High],DH_Low [Low],DH_Close [Close],DH_Volume [Volume]
,DH_CloseChange [CloseChange],DH_CloseChangePct [CloseChangePct]
FROM Supra.dbo.tblDataHarvester_Config x
LEFT JOIN Supra.dbo.tblDataHarvester y ON y.DH_DHC_RecordID = x.DHC_RecordID
