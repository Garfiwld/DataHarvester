CREATE VIEW dbo.viwDataHarvester AS
SELECT 
	x.DHC_RecordID [ID]
	,x.DHC_Name [Name]
	,x.DHC_Symbol [Symbol]
	,x.DHC_Exchange [Exchange]
	,x.DHC_Interval [Interval]
	,y.DH_TimeStamp [DateTime]
	,CAST(y.DH_TimeStamp AS DATE) [Date]
	,CONVERT(DATE,LEFT(CAST(y.DH_TimeStamp AS DATE),7)+'-01') [YM1]
	,y.DH_Open [Open]
	,y.DH_High [High]
	,y.DH_Low [Low]
	,y.DH_Close [Close]
	,y.DH_CloseChange [CloseChange]
	,y.DH_CloseChangePct [CloseChangePct]
	,y.DH_Volume [Volume]
	,x.DHC_Currency [Currency]
	,x.DHC_Unit [Unit]
	,z.DH_Close [1USD_THB]
	,y.DH_Open*z.DH_Close [Open_THB]
	,y.DH_High*z.DH_Close [High_THB]
	,y.DH_Low*z.DH_Close [Low_THB]
	,y.DH_Close*z.DH_Close [Close_THB]
	,y.DH_CloseChange*z.DH_Close [CloseChange_THB]
FROM Supra.dbo.tblDataHarvester_Config x
LEFT JOIN Supra.dbo.tblDataHarvester y ON y.DH_DHC_RecordID = x.DHC_RecordID
LEFT JOIN Supra.dbo.tblDataHarvester z ON z.DH_DHC_RecordID = 1 AND CAST(z.DH_TimeStamp AS DATE) = CAST(y.DH_TimeStamp AS DATE) AND y.DH_DHC_RecordID NOT IN (1,2)
